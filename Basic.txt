ssh -i mylab.pem remoteuser@remotehost	

//user logging out of an SSH session

//create private & public key for key based auth
// private and public keys are saved in the   /home/user/.ssh/id_rsa and /home/user/.ssh/id_rsa.pub files
ssh-keygen	

//create set of SSH keys with passphrase protection
ssh-keygen -f .ssh/key2

//share the pubic key
ssh-copy-id -i .ssh/key-with-pass.pub user@remotehost

//avoid interactively typing in the passphrase while logging in with SSH
eval $(ssh-agent)
